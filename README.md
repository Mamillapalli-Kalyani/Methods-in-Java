# Methods-in-Java
public class Main
{
  public static void main(String args[])
  {
    System.out.println(add("Hello"," World"));
  }
  public static int add(int a,int b)
  {
    return a+b;
  }
  public static float add(float a, float b)
  {
    return a+b;
  }
  public static String add(String a,String b)
  {
    return a+b;
  }
}
